package com.example;

import java.lang.reflect.Field;
import java.lang.reflect.Method;

public class ReflectionExample {
    public static void main(String[] args) throws Exception {
        // 1. Obținem referința clasei
        Class<?> clazz = Persoana.class;

        // 2. Creăm o instanță a clasei folosind Reflection
        Persoana persoana = (Persoana) clazz.getDeclaredConstructor().newInstance();

        // 3. Accesăm și modificăm câmpul privat "nume"
        Field field = clazz.getDeclaredField("nume");
        field.setAccessible(true); // Permite accesul la câmp privat
        field.set(persoana, "Ionel"); // Modifică valoarea

        // 4. Apelăm metoda privată "afiseazaMesaj"
        Method metoda = clazz.getDeclaredMethod("afiseazaMesaj");
        metoda.setAccessible(true); // Permite accesul la metoda privată
        metoda.invoke(persoana); // Apelează metoda
    }
}
