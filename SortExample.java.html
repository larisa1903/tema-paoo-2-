import java.util.List;
import java.util.stream.Collectors;

public class SortExample {
    public static void main(String[] args) {
        List<String> nume = List.of("Maria", "Andrei", "Ion", "Elena");

        // Sortare alfabetică
        List<String> sortat = nume.stream()
                .sorted()
                .collect(Collectors.toList());

        System.out.println(sortat); // [Andrei, Elena, Ion, Maria]

        // Sortare inversă
        List<String> sortatDesc = nume.stream()
                .sorted((a, b) -> b.compareTo(a))
                .collect(Collectors.toList());

        System.out.println(sortatDesc); // [Maria, Ion, Elena, Andrei]
    }
}
